#include <QApplication>
#include <QWidget>

#include <QLabel>
#include <QLineEdit>
#include <QPushButton>

#include <QVBoxLayout>
#include <QHBoxLayout>
#include <QGridLayout>
#include <QGroupBox>
#include <QTextEdit>
#include <QFormLayout>

#include <QComboBox>
#include <QSpinBox>


class Window : public QWidget
{
	public : 
		Window(QWidget* parent = nullptr) : QWidget(parent)
	{
		setWindowTitle("Login");
		setGeometry(250,250,500,500);
	
		line1_label = new QLabel("line 1: ");
		line2_label = new QLabel("line 2: ");
		line3_label = new QLabel("line 3: ");
		line4_label = new QLabel("line 4: ");
		line5_label = new QLabel("line 5: ");
		line6_label = new QLabel("line 6: ");
		

		line1_edit = new QLineEdit();
		line2_edit = new QLineEdit();
		line3_edit = new QLineEdit();
		line4_edit = new QLineEdit();
		line5_edit = new QLineEdit();
		line6_edit = new QLineEdit();
		form_layout = new QFormLayout();

		combo_box = new QComboBox();
		combo_box->addItem("Item 1");
		combo_box->addItem("Item 2");
		combo_box->addItem("Item 3");
		spin_box = new QSpinBox();
	//=====================================
	//Main layout
		main_layout = new QVBoxLayout();
		setLayout(main_layout);
		
	//=====================================
	
	grid_group_box = new QGroupBox();
	grid_group_box->setTitle("Login Form");

	grid_layout = new QGridLayout();
	grid_group_box->setLayout(grid_layout);
	
	main_layout->addWidget(grid_group_box);
	grid_text_edit = new QTextEdit();

	grid_layout->addWidget(line1_label,0,0);
	grid_layout->addWidget(line1_edit,0,1);
	grid_layout->addWidget(line2_label,1,0);
	grid_layout->addWidget(line2_edit,1,1);
	grid_layout->addWidget(line3_label,2,0);
	grid_layout->addWidget(line3_edit,2,1);

	grid_layout->addWidget(grid_text_edit,0,2,3,1);
	//=====================================
	grid_group_box2 = new QGroupBox();
	form_layout->addRow(line4_label,line4_edit);
	form_layout->addRow(line5_label,combo_box);
	form_layout->addRow(line6_label,spin_box);
	grid_group_box2->setTitle("Form Layout");
	grid_group_box2->setLayout(form_layout);
	main_layout->addWidget(grid_group_box2);
	//=====================================
	//Connections
	connect(combo_box,QOverload<int>::of(&QComboBox::currentIndexChanged),this,&Window::on_index_changed);
	connect(spin_box,QOverload<int>::of(&QSpinBox::valueChanged),this,&Window::on_spin_change);
	

}


	void on_index_changed(int index){
	auto text = combo_box->itemText(index);
	printf("%s \n", text.toStdString().c_str());
	}
	
	void on_spin_change(int value){
		printf("%d\n",value);
	}
	
	
	private:
	QVBoxLayout* main_layout;
	
	QGridLayout* grid_layout;
	QGroupBox* grid_group_box;
	QGroupBox* grid_group_box2;
	QTextEdit* grid_text_edit;
	QFormLayout* form_layout;

	QComboBox* combo_box;
	QSpinBox* spin_box;



	QLabel* line1_label;
	QLabel* line2_label;
	QLabel* line3_label;
	QLabel* line4_label;
	QLabel* line5_label;
	QLabel* line6_label;
	
	QLineEdit* line1_edit;
	QLineEdit* line2_edit;
	QLineEdit* line3_edit;
	QLineEdit* line4_edit;
	QLineEdit* line5_edit;
	QLineEdit* line6_edit;
	

};

int main(int argc, char* argv[]){
	QApplication application(argc,argv);
	Window window;
	window.show();
	return application.exec();
}
